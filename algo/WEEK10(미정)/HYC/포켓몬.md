import java.util.*;
class Solution {
    public int solution(int[] nums) {
        int len = nums.length;
        HashSet<Integer> m = new HashSet<>();
        for (int i = 0; i < len; i++){
            m.add(nums[i]);
        }
        int answer = m.size() > len/2 ? len/2 : m.size();
        return answer;
    }
}
